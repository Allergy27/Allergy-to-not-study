        if(x>right)right=x;
    }
    right++;
    while(left<right){//äºŒåˆ†ç­”æ¡ˆ
        ll mid=(left+right+1)>>1;
        ll sum=0;
        for(auto &x:qwq)sum+=min(x,mid);
        if(sum>=k)right=mid-1;
        else left=mid;
    }
    ll sum=0;
    for(auto &x:qwq){//æ¨¡æ‹Ÿä¸€ä¸‹åƒçš„è¿‡ç¨‹
        sum+=min(x,left);
        x-=min(x,left);
    }
    ll res=k-sum;//è¿˜è¦åƒçš„æ•°é‡
    for(i=0;i<n&&res;++i){
        if(qwq[i]){
            --qwq[i];
            --res;
        }
    }
    for(auto &x:qwq)cout<<x<<" ";
}
